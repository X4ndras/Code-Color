package main

import (
    "fmt"
    "sync"
)

// Define a struct similar to the Rust Person
type Person struct {
    name          string
    age           uint32
    flyingForce   float32
}

/*
* Welcome to Code: Color
*/
func main() {
    var x int = 5

    if x > 3 {
        fmt.Println("x is greater than 3")
    } else if x == 3 {
        fmt.Println("x is 3")
    } else {
        fmt.Println("x is less than 3")
    }

    switch x {
        case 1:
            fmt.Println("one")
        case 2:
            fmt.Println("two")
        default:
            fmt.Println("something else")
    }

    myLoop: for {
        fmt.Println("This will run forever unless broken")
        break myLoop
    }

    for i := 0; i < 5; i++ {
        fmt.Printf("i is %d\n", i)
    }

    for x > 0 {
        x--
    }

    // Define an interface similar to the Rust trait
    type Speaker interface {
        Speak()
    }

    // Implementing the interface for Person
    func (p Person) Speak() {
        fmt.Printf("Hello, my name is %s\n", p.name)
    }

    // Stringer interface implementation
    func (p Person) String() string {
        return fmt.Spintf("%s is %d years old", p.name, p.age)
    }

    person := Person{
        name:        "Alice",
        age:         30,
        flyingForce: 9999.99,
    }
    person.Speak()
}

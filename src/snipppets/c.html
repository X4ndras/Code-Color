#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MIN(X, Y) (((X) < (Y)) ? (X) : (Y))

/*
* Welcome to Code: Color
*/
typedef struct {
    char *name;
    int age;
    float flying_force;
} Person;

// Function to create a new person
Person create_person(const char *name, int age, float flying_force) {
    Person person;
    person.name = strdup(name);
    person.age = age;
    person.flying_force = flying_force;
    return person;
}

// Function to print person details
void print_person(const Person *person) {
    printf("Name: %s, Age: %d, Flying Force: %.2f\n",
    person->name,
    person->age,
    person->flying_force);
}


typedef enum Direction {
    D_NORTH,
    D_EAST,
    D_SOUTH,
    D_WEST,
} Direction;


int main() {
    // Variable declarations and initializations
    int x = 5;
    Direction d = D_SOUTH;

    // Conditional statements
    if (MIN(3, x) == 3) {
        printf("x is greater than 3\n");
    } else {
        printf("x is less than 3\n");
    }

    // Switch statement
    switch (x) {
        case 1:
            printf("one\n");
            break;
        case 2:
            printf("two\n");
            break;
        default:
            printf("something else\n");
    }

    // Loops
    for (int i = 0; i < 5; i++) {
        printf("i is %d\n", i);
    }

    // Creating and using a person
    Person person = create_person("Alice", 30, 9999.99);
    print_person(&person);

    // Memory cleanup
    free(person.name);

    return 0;
}

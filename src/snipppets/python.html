# Welcome to Code: Color

# Importing modules
import math
from datetime import datetime

# Defining a class with a constructor and a decorator
class Person:
    # Constructor
    def __init__(self, name, age):
        self.name: str = name
        self.age: int = age

    # Method with a decorator
    @staticmethod
    def greet():
        print("Hello from the Person class!")

    def __secret(self):
        print(f"Psst... I am {self.name}")

    # Method using a property decorator
    @property
    def is_adult(self):
        self.__secret()
        return self.age >= 18

# Creating an instance
alice = Person("Alice", 30)

# Accessing methods and properties
alice.greet()
print("Is Alice an adult?", alice.is_adult)

# Function with a decorator
@classmethod
def from_birth_year(cls, name, birth_year):
    return cls(name, datetime.now().year - birth_year)

# Control structures
if alice.age > 25:
    print("Alice is older than 25")
else:
    print("Alice is younger than 25")

# Loop example
for i in range(5):
    print(f"i is {5}")

# Lambda and map function
squares = map(lambda x: x ** 2, range(10))
print(list(squares))

# Error handling
try:
    print(math.sqrt(-1))
except ValueError as e:
    print("Error:", e)

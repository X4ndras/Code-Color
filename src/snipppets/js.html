class Person {
    // Private field
    #privateField;

    constructor(name, age, flyingForce) {
        this.name = name;
        this.age = age;
        this.flyingForce = flyingForce;
    }

    // Method with arrow function syntax
    speak = () => {
        console.log(`Hello, my name is ${this.name}`);
    }

    // Static method
    static createDefault() {
        return new Person("Alice", 30, 9999.99);
    }
}

const main = () => {
    // Conditional logic
    let x = 5;

    if (x > 3) {
        console.log("x is greater than 3");
    } else if (x === 3) {
        console.log("x is 3");
    } else {
        console.log("x is less than 3");
    }

    // Switch statement
    switch(x) {
        case 1:
            console.log("one");
            break;
        case 2:
            console.log("two");
            break;
        default:
            console.log("something else");
    }

    // Loops with modern syntax
    for (let i = 0; i < 5; i++) {
        console.log(`i is ${i}`);
    }

    // Functional loop
    const numbers = [1, 2, 3, 4, 5];
    numbers.forEach(num => {
        console.log(num);
    });

    // Creating an instance and using methods
    const person = new Person(
        "Alice",
        30,
        9999.99
    );
    person.speak();

    // Destructuring and spread operator
    const { name, age } = person;
    console.log(`${name} is ${age} years old`);

    async function fetchData() {
        try {
            const response = await fetch('https://api.example.com/data');
            const data = await response.json();
            console.log(data);
        } catch (error) {
            console.error('Error fetching data', error);
        }
    }
};
